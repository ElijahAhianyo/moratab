```python
def append_simple_footnotes(text):
    for footnote in re.finditer('\\[\\^([^\\]]+)\\]', text):
        ref = footnote.group(1)
        if not '[^{0}]:'.format(ref) in text:
            text += '\n[^{0}]: {0}'.format(ref)
    return text
```
